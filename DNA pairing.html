function pairElement(str) { let pairs = []; const strArr = str.split(""); strArr.map(elem => { switch (elem) { case 'C': pairs.push(['C', 'G']); break; case 'G': pairs.push(['G', 'C']); break; case 'A': pairs.push(['A', 'T']); break; case 'T': pairs.push(['T', 'A']); break; } }) return pairs; } pairElement("GCG");
